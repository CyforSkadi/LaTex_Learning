\documentclass{ctexart}

\usepackage{amsmath}
\usepackage{amssymb}

\begin{document}
	%gather(gather*)环境多行公式排版，需要引用amsmath和amssymb宏包
	
	%带编号
	\begin{gather}	
		a + b = b + a \\
		ab ba
	\end{gather}
	
	%不带编号
	\begin{gather*}
		a + b = b + a \\
		ab ba
	\end{gather*}

	%在\\前用\notag阻止编号
	\begin{gather}
		3^2 + 4^2 = 5^2 \notag \\
		3^2 + 4^2 = 5^2 \notag \\
		3^2 + 4^2 = 5^2 
	\end{gather}

	%align和align*环境公式实现多个对齐(用&对齐)
	\begin{align}
		x &= t + \cos t + 1 \\
		y &= 2\sin t 
	\end{align}

	%split环境(在equation环境中)实现一个公式多行排版
	\begin{equation}
		\begin{split}
			\cos 2x &= \cos^2 x - \sin^2 x \\
			&= 2\cos^2 x -1 
		\end{split}
	\end{equation}

	%cases环境(在equation环境中)实现分段函数
	%每行公式中使用&分隔为两部分
	%通常表示值和后面的条件
	\begin{equation}
		D(x) = 
		\begin{cases}
			1,& \text{如果} x \in \mathbb{Q}; \\	%\in输出属于符号，\mathbb输出花体字符，需要引用amssymb宏包
			0,& \text{如果} x \in \mathbb{R};
		\end{cases}
	\end{equation}
	
\end{document}
