\documentclass{ctexart}

%\newcommand-定义命令
%命令只能由字母组成，不能以\end开头
%\newcommand{命令}[参数个数][首参数默认值]{具体定义}
\newcommand{\PRC}{People's Republic of \emph{China}}	%自定义字符替换命令

%参数个数为1~9个，使用时用#1，#2，...，#9表示
\newcommand{\loves}[2]{#1 喜欢 #2}
\newcommand{\loved}[2]{#2 被 #1 喜欢}

%指定参数同时可以指定首个参数的默认值，则该命令的第一个参数就成为可选的参数(用中括号指定)
\newcommand{\love}[3][喜欢]{#2#1#3}

%\renewcommand-重定义命令
%与\newcommand 命令作用和语法相同

\begin{document}
	\PRC
	
	\loves{蛋挞}{璇砸}
	
	\loved{蛋挞}{璇砸}
	
	\love{蛋挞}{璇砸}
	\love[最爱]{蛋挞}{璇砸}
	
\end{document}
