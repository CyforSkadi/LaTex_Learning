\documentclass{ctexart}

\usepackage{amsmath}

\begin{document}
	%矩阵环境，用&分隔列，用\\分隔行
	\[
	\begin{matrix}	%使用matrix环境，需要引用amsmath宏包
		0 & 1 \\
		1 & 0
	\end{matrix}	\quad
	\begin{pmatrix}	%pmatrix环境，加小括号
		0 & 1 \\
		1 & 0
	\end{pmatrix}	\quad
	\begin{bmatrix}	%bmatrix环境，加中括号
		0 & 1 \\
		1 & 0
	\end{bmatrix}	\quad
	\begin{Bmatrix}	%Bmatrix环境，加大括号
		0 & 1 \\
		1 & 0
	\end{Bmatrix}	\quad
	\begin{vmatrix}	%vmatrix环境，加单竖线
		0 & 1 \\
		1 & 0
	\end{vmatrix}	\quad
	\begin{Vmatrix}	%Vmatrix环境，加双竖线
		0 & 1 \\
		1 & 0
	\end{Vmatrix}	\quad
	\]
	
	%常用省略号：\dots \ddots \vdots
	\[
	A = 
	\begin{bmatrix}
		a_{11} & \dots & a_{1n}	\\
		& \ddots & \vdots \\
		0 & & a_{nn}
	\end{bmatrix}_{n \times n}
	\]
	
	%分块矩阵(矩阵嵌套)
	\[
	\begin{pmatrix}
		\begin{matrix}	1&0\\0&1 \end{matrix}
		& \text{\Large 0}	\\
		\text{\Large 1}
		& \begin{matrix} 1&0\\0&-1 \end{matrix}

	\end{pmatrix}
	\]
	
	%行内小矩阵(smallmatrix环境)
	复数 $z = (x,y)$ 也可用矩阵
	\begin{math}
		\left(	%需要手动加上左括号
		\begin{smallmatrix}
			x & -y \\ y & x
		\end{smallmatrix}
		\right)	%需要手动加上右括号
	\end{math}来表示。

	%array环境(类似于表格环境tabular)
	\[
	\begin{array}{l|l}
		\frac{1}{2} & 0 \\
		\hline
		0 & -\frac{a}{b}c	\\
	\end{array}
	\]
\end{document}
